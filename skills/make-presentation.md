# Skill: make-presentation

## Triggers
Invoke this skill when the user says any of:
- "haz una presentación"
- "crea un deck"
- "genera slides"
- "make a presentation"
- "make slides"
- "presentation on [topic]"
- "deck sobre [tema]"
- "crea una presentación sobre"
- "slides para"
- "build me a deck"

## What This Skill Does

Generates a professional .pptx presentation using python-pptx with a dark tech theme. Every deck looks like it came from a top-tier tech consultancy — dark navy background, teal accents, electric blue highlights, embedded logo.

After generating, sends the file to the user via Telegram using the `[SEND_FILE:...]` marker.

---

## How to Execute

1. Parse Gonzalo's request to extract:
   - **Topic / title** of the presentation
   - **Slides requested** (if specified) or infer a logical structure (5-8 slides typical)
   - **Content** for each slide (bullet points, table data, section headers)

2. Write a Python script to `/tmp/[topic-slug]-presentation.py` using the master template below.

3. Run it: `python3 /tmp/[topic-slug]-presentation.py`

4. Verify the file was created: `ls -lh /tmp/[topic-slug]-presentation.pptx`

5. Save to Brain Vault: `bash /Users/opoclaw1/claudeclaw/scripts/brain-save.sh "/tmp/[topic-slug]-presentation.pptx" "Negocio"`

6. Send to Gonzalo: include `[SEND_FILE:/tmp/[topic-slug]-presentation.pptx|[Topic] — Deck listo]` in your response.

```python
#!/usr/bin/env python3
"""
OpoClaw Presentation Generator
Dark tech theme — consulting-grade quality
Generated by make-presentation skill
"""

import os
from pptx import Presentation
from pptx.util import Inches, Pt, Emu
from pptx.dml.color import RGBColor
from pptx.enum.text import PP_ALIGN
from datetime import datetime

# ─── COLOR PALETTE ────────────────────────────────────────────────────────────
BG         = RGBColor(0x0A, 0x0E, 0x1A)
BG_CARD    = RGBColor(0x11, 0x18, 0x27)
BG_ALT     = RGBColor(0x1A, 0x23, 0x32)
TEAL       = RGBColor(0x0D, 0x94, 0x88)
TEAL_LIGHT = RGBColor(0x14, 0xB8, 0xA6)
BLUE       = RGBColor(0x3B, 0x82, 0xF6)
WHITE      = RGBColor(0xFF, 0xFF, 0xFF)
TEXT       = RGBColor(0xE2, 0xE8, 0xF0)
MUTED      = RGBColor(0x94, 0xA3, 0xB8)
BORDER     = RGBColor(0x1E, 0x3A, 0x4A)

# ─── CONSTANTS ────────────────────────────────────────────────────────────────
SLIDE_W   = Inches(13.33)
SLIDE_H   = Inches(7.5)
LOGO_PATH = "/Users/opoclaw1/claudeclaw/workspace/opoclaw-logo-hd.png"
OUTPUT    = "/tmp/TOPIC-presentation.pptx"  # <-- change TOPIC per deck


def new_prs():
    prs = Presentation()
    prs.slide_width = SLIDE_W
    prs.slide_height = SLIDE_H
    return prs


def fill_background(slide, color=BG):
    background = slide.background
    fill = background.fill
    fill.solid()
    fill.fore_color.rgb = color


def add_logo(slide, right_margin=Inches(0.25), top_margin=Inches(0.2), width=Inches(1.2)):
    if not os.path.exists(LOGO_PATH):
        return
    try:
        left = SLIDE_W - width - right_margin
        slide.shapes.add_picture(LOGO_PATH, left, top_margin, width=width)
    except Exception:
        pass


def add_text_box(slide, text, left, top, width, height,
                 font_size=18, color=TEXT, bold=False, align=PP_ALIGN.LEFT,
                 font_name="Calibri"):
    txBox = slide.shapes.add_textbox(left, top, width, height)
    tf = txBox.text_frame
    tf.word_wrap = True
    p = tf.paragraphs[0]
    p.alignment = align
    run = p.add_run()
    run.text = text
    run.font.size = Pt(font_size)
    run.font.color.rgb = color
    run.font.bold = bold
    run.font.name = font_name
    return txBox


def add_line(slide, left, top, width, color=TEAL, height_px=3):
    line = slide.shapes.add_shape(1, left, top, width, Pt(height_px))
    line.fill.solid()
    line.fill.fore_color.rgb = color
    line.line.fill.background()


def add_rect(slide, left, top, width, height, fill_color=BG_CARD, line_color=None):
    shape = slide.shapes.add_shape(1, left, top, width, height)
    shape.fill.solid()
    shape.fill.fore_color.rgb = fill_color
    if line_color:
        shape.line.color.rgb = line_color
    else:
        shape.line.fill.background()
    return shape


# ─── SLIDE BUILDERS ───────────────────────────────────────────────────────────

def make_cover_slide(prs, title, subtitle, date=None):
    """Cover: big title, subtitle, teal accent bar, logo, date."""
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    fill_background(slide)
    add_rect(slide, Inches(0), Inches(2.8), Inches(0.12), Inches(2.0), TEAL)
    add_text_box(slide, title,
        left=Inches(0.5), top=Inches(2.4), width=Inches(10), height=Inches(1.4),
        font_size=42, color=WHITE, bold=True, align=PP_ALIGN.LEFT)
    add_line(slide, Inches(0.5), Inches(3.9), Inches(9), TEAL, 3)
    add_text_box(slide, subtitle,
        left=Inches(0.5), top=Inches(4.1), width=Inches(9), height=Inches(0.6),
        font_size=20, color=TEAL_LIGHT, align=PP_ALIGN.LEFT)
    date_str = date or datetime.now().strftime("%B %Y")
    add_text_box(slide, date_str,
        left=Inches(0.5), top=Inches(6.6), width=Inches(4), height=Inches(0.4),
        font_size=12, color=MUTED, align=PP_ALIGN.LEFT)
    add_text_box(slide, "opoclaw.com",
        left=Inches(9.5), top=Inches(6.6), width=Inches(3.5), height=Inches(0.4),
        font_size=11, color=MUTED, align=PP_ALIGN.RIGHT)
    add_logo(slide, right_margin=Inches(0.4), top_margin=Inches(0.25), width=Inches(1.4))
    return slide


def make_section_slide(prs, title, body=None):
    """Section header: full-width teal band top, centered large title."""
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    fill_background(slide)
    add_rect(slide, Inches(0), Inches(0), SLIDE_W, Inches(1.1), TEAL)
    add_text_box(slide, "SECTION",
        left=Inches(0.5), top=Inches(0.15), width=Inches(6), height=Inches(0.5),
        font_size=11, color=WHITE, bold=True, align=PP_ALIGN.LEFT)
    add_text_box(slide, title,
        left=Inches(1), top=Inches(2.5), width=Inches(11.33), height=Inches(1.5),
        font_size=38, color=WHITE, bold=True, align=PP_ALIGN.CENTER)
    if body:
        add_text_box(slide, body,
            left=Inches(2), top=Inches(4.2), width=Inches(9.33), height=Inches(0.8),
            font_size=18, color=MUTED, align=PP_ALIGN.CENTER)
    add_logo(slide, right_margin=Inches(0.3), top_margin=Inches(0.15), width=Inches(1.0))
    return slide


def make_content_slide(prs, title, bullets):
    """Content: title + teal underline + teal-dotted bullet list."""
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    fill_background(slide)
    add_text_box(slide, title,
        left=Inches(0.5), top=Inches(0.35), width=Inches(10), height=Inches(0.8),
        font_size=28, color=WHITE, bold=True, align=PP_ALIGN.LEFT)
    add_line(slide, Inches(0.5), Inches(1.15), Inches(12.33), TEAL, 2)
    bullet_top = Inches(1.4)
    bullet_gap = Inches(0.85)
    for i, bullet in enumerate(bullets):
        dot = slide.shapes.add_shape(1,
            Inches(0.5), bullet_top + i * bullet_gap + Inches(0.12),
            Inches(0.12), Inches(0.12))
        dot.fill.solid()
        dot.fill.fore_color.rgb = TEAL
        dot.line.fill.background()
        add_text_box(slide, bullet,
            left=Inches(0.75), top=bullet_top + i * bullet_gap,
            width=Inches(11.6), height=Inches(0.7),
            font_size=18, color=TEXT, align=PP_ALIGN.LEFT)
    add_logo(slide, right_margin=Inches(0.3), top_margin=Inches(0.25), width=Inches(1.1))
    return slide


def make_two_column_slide(prs, title, left_header, left_items, right_header, right_items):
    """Two-column slide with teal divider."""
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    fill_background(slide)
    add_text_box(slide, title,
        left=Inches(0.5), top=Inches(0.35), width=Inches(12), height=Inches(0.8),
        font_size=28, color=WHITE, bold=True, align=PP_ALIGN.LEFT)
    add_line(slide, Inches(0.5), Inches(1.15), Inches(12.33), TEAL, 2)
    # Teal vertical divider
    add_rect(slide, Inches(6.6), Inches(1.35), Inches(0.05), Inches(5.5), TEAL)
    # Left column
    add_text_box(slide, left_header,
        left=Inches(0.5), top=Inches(1.4), width=Inches(5.8), height=Inches(0.5),
        font_size=16, color=TEAL_LIGHT, bold=True)
    for i, item in enumerate(left_items):
        add_text_box(slide, f"• {item}",
            left=Inches(0.5), top=Inches(2.0 + i * 0.75), width=Inches(5.8), height=Inches(0.6),
            font_size=16, color=TEXT)
    # Right column
    add_text_box(slide, right_header,
        left=Inches(6.85), top=Inches(1.4), width=Inches(6.0), height=Inches(0.5),
        font_size=16, color=TEAL_LIGHT, bold=True)
    for i, item in enumerate(right_items):
        add_text_box(slide, f"• {item}",
            left=Inches(6.85), top=Inches(2.0 + i * 0.75), width=Inches(6.0), height=Inches(0.6),
            font_size=16, color=TEXT)
    add_logo(slide, right_margin=Inches(0.3), top_margin=Inches(0.25), width=Inches(1.1))
    return slide


def make_table_slide(prs, title, headers, rows):
    """Table slide: teal header row, dark alternating rows."""
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    fill_background(slide)
    add_text_box(slide, title,
        left=Inches(0.5), top=Inches(0.35), width=Inches(10), height=Inches(0.8),
        font_size=28, color=WHITE, bold=True, align=PP_ALIGN.LEFT)
    add_line(slide, Inches(0.5), Inches(1.15), Inches(12.33), TEAL, 2)

    num_cols = len(headers)
    num_rows = len(rows) + 1
    table = slide.shapes.add_table(
        num_rows, num_cols,
        Inches(0.5), Inches(1.4),
        Inches(12.33), Inches(0.5 * num_rows + 0.3)
    ).table

    def style_cell(cell, text, bg_color, text_color=WHITE, bold=False, font_size=14):
        cell.text = text
        cell.fill.solid()
        cell.fill.fore_color.rgb = bg_color
        p = cell.text_frame.paragraphs[0]
        p.alignment = PP_ALIGN.LEFT
        run = p.runs[0] if p.runs else p.add_run()
        run.text = text
        run.font.size = Pt(font_size)
        run.font.color.rgb = text_color
        run.font.bold = bold
        run.font.name = "Calibri"

    for col_idx, header in enumerate(headers):
        style_cell(table.cell(0, col_idx), header, TEAL, WHITE, bold=True, font_size=14)
    for row_idx, row_data in enumerate(rows):
        bg = BG_CARD if row_idx % 2 == 0 else BG_ALT
        for col_idx, cell_text in enumerate(row_data):
            style_cell(table.cell(row_idx + 1, col_idx), str(cell_text), bg, TEXT, font_size=13)

    add_logo(slide, right_margin=Inches(0.3), top_margin=Inches(0.25), width=Inches(1.1))
    return slide


def make_closing_slide(prs, title="Thank You", subtitle=None):
    """Closing: centered minimal with teal footer band."""
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    fill_background(slide)
    add_rect(slide, Inches(0), Inches(6.8), SLIDE_W, Inches(0.7), TEAL)
    add_text_box(slide, title,
        left=Inches(1), top=Inches(2.5), width=Inches(11.33), height=Inches(1.2),
        font_size=44, color=WHITE, bold=True, align=PP_ALIGN.CENTER)
    if subtitle:
        add_text_box(slide, subtitle,
            left=Inches(2), top=Inches(3.9), width=Inches(9.33), height=Inches(0.7),
            font_size=20, color=TEAL_LIGHT, align=PP_ALIGN.CENTER)
    add_text_box(slide, "www.opoclaw.com  |  your@email.com",
        left=Inches(3), top=Inches(6.88), width=Inches(7.33), height=Inches(0.4),
        font_size=13, color=WHITE, align=PP_ALIGN.CENTER)
    add_logo(slide, right_margin=Inches(0.4), top_margin=Inches(0.3), width=Inches(1.4))
    return slide

def build_presentation():
    prs = new_prs()

    # SLIDE 1: Cover
    make_cover_slide(
        prs,
        title="DECK TITLE HERE",
        subtitle="Subtitle or author line",
        date=datetime.now().strftime("%B %Y")

    # SLIDE 2: Section header
    make_section_slide(
        prs,
        title="Section Title",
        body="Optional subtitle or description"
    )

    # SLIDE 3: Content with bullets
    make_content_slide(
        prs,
        title="Key Points",
        bullets=[
            "First important point goes here",
            "Second point with supporting detail",
            "Third insight or finding",
            "Fourth takeaway for the audience",
        ]
    )

    # SLIDE 4: Two-column comparison
    make_two_column_slide(
        prs,
        title="Comparison",
        left_header="Option A",
        left_items=["Fast to deploy", "Lower upfront cost", "Scalable architecture"],
        right_header="Option B",
        right_items=["Higher control", "Better long-term ROI", "Custom integrations"]
    )

    # SLIDE 5: Table
    make_table_slide(
        prs,
        title="Data Overview",
        headers=["Column 1", "Column 2", "Column 3"],
        rows=[
            ["Row 1 A", "Row 1 B", "Row 1 C"],
            ["Row 2 A", "Row 2 B", "Row 2 C"],
            ["Row 3 A", "Row 3 B", "Row 3 C"],
        ]
    )

    # SLIDE 6: Closing
    make_closing_slide(
        prs,
        title="Thank You",
        subtitle="OpoClaw — Built to Scale"

    prs.save(OUTPUT)
    print(f"Saved: {OUTPUT}")


if __name__ == "__main__":
    build_presentation()
```

## Slide Types Available

| Function | When to use |
|---|---|
| `make_cover_slide(prs, title, subtitle, date)` | Always first slide |
| `make_section_slide(prs, title, body)` | Divider between major sections |
| `make_content_slide(prs, title, bullets)` | Main content — up to 6 bullets |
| `make_two_column_slide(prs, title, left_header, left_items, right_header, right_items)` | Comparisons, pros/cons, before/after |
| `make_table_slide(prs, title, headers, rows)` | Data, metrics, team lists |
| `make_closing_slide(prs, title, subtitle)` | Always last slide |

## Design Principles

- Dark navy background on every slide, always. No white slides.
- Teal (#0D9488) is the primary accent — underlines, dots, header rows, bars.
- White for primary headers. Light gray (#E2E8F0) for body text.
- Calibri font throughout (Arial fallback).
- Logo top-right on every slide (silent fallback if file missing).
- Minimum 5 slides, maximum ~15. Quality over quantity.
- No walls of text — bullets, tables, and visual dividers only.
- Each slide should communicate one clear idea.
